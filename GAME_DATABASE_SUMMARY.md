## 🔧 HOW TO USE

### Basic Import
```typescript
import { GameDatabase } from './data';

// Access any content category
const weapons = GameDatabase.weapons;
const spells = GameDatabase.spells;
const bosses = GameDatabase.bosses;
const locations = GameDatabase.locations;
```

### Use Utility Functions
```typescript
import {
  getWeaponsByType,
  getSpellsBySchool,
  getBossesByTier,
  getLocationsByRegion,
  getFastTravelLocations
} from './data';

// Get specific items
const swords = getWeaponsByType('Sword');
const sorcery = getSpellsBySchool('Sorcery');
const legendaryBosses = getBossesByTier('legendary');
```

### Integration with Screens
```typescript
// In ItemCrafting.tsx
import { GameDatabase } from '../data';

const allWeapons = Object.values(GameDatabase.weapons).flat();
```

## 🎯 INTEGRATION EXAMPLES

### Example 1: ItemCrafting Screen
```typescript
import { GameDatabase } from '../data';

export function ItemCraftingScreen() {
  const recipes = [
    ...GameDatabase.weapons.SWORDS,
    ...GameDatabase.weapons.GREATSWORDS,
    ...GameDatabase.armor.CHESTS,
  ];
  
  return (
    <FlatList
      data={recipes}
      renderItem={({ item }) => <RecipeCard recipe={item} />}
    />
  );
}
```

### Example 2: DungeonCrawler Screen
```typescript
import { getDungeonsByType } from '../data';

export function DungeonSelection() {
  const dungeons = getDungeonsByType('dungeon');
  const raids = getDungeonsByType('raid');
  
  return (
    <View>
      <DungeonList dungeons={dungeons} />
      <DungeonList dungeons={raids} />
    </View>
  );
}
```

### Example 3: World Map Screen
```typescript
import { getLocationsByRegion } from '../data';

export function WorldMap() {
  const limgraveLocations = getLocationsByRegion('Limgrave');
  
  return (
    <Map>
      {limgraveLocations.map(location => (
        <MapMarker key={location.id} location={location} />
      ))}
    </Map>
  );
}
```
