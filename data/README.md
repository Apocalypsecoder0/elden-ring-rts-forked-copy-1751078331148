## 🔧 Usage Examples

### Import Game Database
```typescript
import { GameDatabase, GameStatistics } from './data';

// Get all weapons
const weapons = GameDatabase.weapons;

// Get bosses by tier
const legendaryBosses = GameDatabase.bosses.legendary;

// Get spells by school
const sorcerySpells = GameDatabase.spells.SORCERY;

// Get locations
const kingdoms = GameDatabase.locations.kingdoms;
```

### Use Utility Functions
```typescript
import { 
  getWeaponsByType,
  getSpellsBySchool,
  getBossesByTier,
  getLocationsByRegion,
  getFastTravelLocations
} from './data';

// Get all sword-type weapons
const swords = getWeaponsByType('Sword');

// Get sorcery spells
const sorcery = getSpellsBySchool('Sorcery');

// Get legendary bosses
const legendaryBosses = getBossesByTier('legendary');

// Get all Liurnia locations
const liurniaLocations = getLocationsByRegion('Liurnia of the Lakes');

// Get fast travel enabled locations
const fastTravelPoints = getFastTravelLocations();
```
